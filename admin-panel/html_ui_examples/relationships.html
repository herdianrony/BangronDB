PHP Code Example

// Populate orders with user
$orders = $db->orders
    ->find()
    ->populate(
        'user_id',
        $db->users,
        ['as' => 'user']
    )
    ->toArray();

// Cross-database populate
$orders = $db->orders
    ->find()
    ->populate(
        'company_id',
        'company_db.companies',
        ['as' => 'company']
    )
    ->toArray();
