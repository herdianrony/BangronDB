PHP Implementation 

// Set Schema Validation
$users->setSchema([
    'username' => [
        'required' => true,
        'type' => 'string',
        'min' => 3,
        'max' => 50
    ],
    'email' => [
        'required' => true,
        'type' => 'string',
        'regex' => '/^[^\s@]+@[^\s@]+\.[^\s@]+$/'
    ],
    'age' => [
        'type' => 'int',
        'min' => 13,
        'max' => 120
    ],
    'role' => [
        'required' => true,
        'type' => 'string',
        'enum' => ['admin', 'editor', 'user', 'viewer']
    ]
]);

// Set Searchable Fields (for encrypted collections)
$users->setSearchableFields(['email'], true);

// Populate Relations
$results = $users->find()
    ->populate('department_id', $db->departments, ['as' => 'department'])
    ->populate('company_id', 'company_db.companies', ['as' => 'company'])
    ->toArray();
