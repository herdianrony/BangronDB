$collection->on('beforeInsert', function($document) {
    // Add timestamps
    if (!isset($document['created_at'])) {
        $document['created_at'] = date('Y-m-d H:i:s');
    }
    
    $document['updated_at'] = $document['created_at'];
    
    return $document;
});
